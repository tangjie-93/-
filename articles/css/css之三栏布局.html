方法2 float+calc

   .left {
      height: 500px;
      width: 300px;
      float: left;
      background: cadetblue;
    }
    .center {
      width: calc(100% - 600px);
      height: 500px;
      display: inline-block;
      /* 将display: inline-block;改为float布局也可以 */
      background: cornflowerblue;
    }
    .right {
      height: 500px;
      width: 300px;
      float: right;
      background: chocolate;
    }

方法3 absolute+margin

    .container {
            height: 100%;
            position: relative;
        }

        .left {
            position: absolute;
            top: 0;
            left: 0;
            width: 300px;
            height: 100%;
            background: cadetblue;
        }

        .right {
            position: absolute;
            top: 0;
            right: 0;
            width: 300px;
            background: cornflowerblue;
            height: 100%;
        }

        .center {
            margin: 0 300px;
            background: cornsilk;
            height: 100%;
        }

        background: darkblue
    }
